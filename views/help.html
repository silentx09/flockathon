EC2

-- Displays a list of EC2 instances.
/aws list-instances

-- Displays the details of an EC2 instance.
/aws show-instance </instanceid>

-- Boots EC2 Instance
/aws start-instance </instanceid>

-- Stops an EC2 instance
/aws stop-instance </instanceid>

-- Terminate/Delete an EC2 instance
/aws delete-instance </instanceid>

-- Reboot an EC2 instance
/aws reboot-instance </instanceid>

S3 Bucket

-- Displays the list of S3 buckets.
/aws list-buckets

-- List files in an S3 bucket
/aws list-files </bucket_name>

Dynamo DB

-- Lists DynamoDB tables.
/aws d-list-tables

-- Displays the status of a table.
/aws d-table-status </table_name>

-- Displays the size(in bytes) of a table.
/aws d-table-size </table_name>

-- Displays the number of items in the table
/aws d-table-count </table_name>

-- Displays the date when the table was created.
/aws d-table-date </table_name>

Api Gateway

--Displays list of APIs
/aws list-apis

--Displays the resources in an API.
/aws list-resources </api_id>

--Displays the stages in an API.
/aws stages </api_id>

Alarms

--Lists down SNS topics
/aws sns-list-topics

--Subscribe to a SNS topic to get alarms.
/aws sns-subscribe <topicArn>
